#include <stdio.h>
#include <locale.h>

int main() {
    setlocale(LC_CTYPE, "ukr");
    int a, b, c;
    printf("Введіть сторони трикутника:\n");
    scanf_s("%d%d%d", &a, &b, &c);
    if (a == b && b != c) printf("Рівнобедренний");
    if (a == b && b == c && c == a) printf("Рівносторонній");
    if (a != b && b != c && c != a) printf("Різносторонній");
    return 0;
}
